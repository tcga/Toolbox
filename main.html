Request data

This template should be a starting point to request data from TCGA, e.g. some CNA data:

var xhr = new XMLHttpRequest();
xhr.open("GET", "https://tcga-data.nci.nih.gov/tcgafiles/ftp_auth/distro_ftpusers/anonymous/tumor/gbm/cgcc/mskcc.org/hg-cgh-244a/cna/mskcc.org_GBM.HG-CGH-244A.Level_3.26.1.0/MSK_251469369165_S01_CGH-v4_10_27Aug08__GCN_V3_A1__CBS_out.txt");
xhr.onload = function () {
    if (xhr.status === 200) {
        console.log(xhr.responseText);
    } else {
        console.error(xhr.status);
    }
};
xhr.send(null);
